  <div class="container">
	<textarea id="plainTextarea"></textarea>
    <pre><code id="yamlTextarea" class="language-yaml"></code></pre>
  </div>
  
  <script>
    document.addEventListener("DOMContentLoaded", function() {
      var plainTextarea = document.getElementById("plainTextarea");
      plainTextarea.addEventListener("input", function() {
		var yamlTextarea = document.getElementById("yamlTextarea");
        var code = this.value;
        var highlightedCode = Prism.highlight(code, Prism.languages.yaml, "yaml");
        yamlTextarea.innerHTML = highlightedCode;
      });
    });
  </script>
